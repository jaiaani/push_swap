#include "push_swap.h"

void	sort_array(int *array, int size)
{
	int	i;
	int	j;
	int	min_index;
	int	temp;

	i = 0;
	while (i < size - 1)
	{
		min_index = i;
		j = i + 1;
		while (j < size)
		{
			if (array[j] < array[min_index])
				min_index = j;
			j++;
		}
		if (min_index != i)
		{
			temp = array[i];
			array[i] = array[min_index];
			array[min_index] = temp;
		}
		i++;
	}
}

void	normalize(t_stack *a)
{
	int		*array;
	int		i;
	t_node	*node;
	int		j;

	i = 0;
	array = malloc(sizeof(int) * a->size);
	if (!array)
		return ;
	node = a->top;
	while (node)
	{
		array[i++] = node->value;
		node = node->next;
	}
	sort_array(array, a->size); // use any sorting method here
	node = a->top;
	while (node)
	{
		j = 0;
		while (j < a->size)
		{
			if (node->value == array[j])
			{
				node->index = j;
				break ;
			}
			j++;
		}
		node = node->next;
	}
	free(array);
}

void	radix_sort(t_stack *a, t_stack *b)
{
	int	i;
	int	j;
	int	max_bits;
	int	size;

	i = 0;
	max_bits = 0;
	size = a->size;
	// Get max number of bits
	while ((size - 1) >> max_bits)
		max_bits++;
	while (i < max_bits)
	{
		j = 0;
		while (j < size)
		{
			if (((a->top->index >> i) & 1) == 0)
				push(a, b);
			else
				rotate(a);
			j++;
		}
		while (b->size > 0)
			push(b, a);
		i++;
	}
}
